ES6+ Modern Syntax

import { apiCall } from './apiService';

const fetchUserData = async (userId) => {
  try {
    const response = await apiCall(`/users/${userId}`);
    const { name, email, profile } = response;
    
    return {
      name,
      email,
      avatar: profile?.avatar ?? 'default.png'
    };
  } catch (error) {
    console.error('Failed to fetch user:', error);
    throw new Error('User not found');
  }
};

// Example usage
fetchUserData(123)
  .then(user => console.log(user))
  .catch(err => console.error(err));

DOM Manipulation

class DynamicUI {
  constructor(containerId) {
    this.container = document.getElementById(containerId);
    this.state = {
      items: [],
      isLoading: false
    };
  }
  
  createElement(tag, classes = [], text = '') {
    const element = document.createElement(tag);
    element.classList.add(...classes);
    element.textContent = text;
    return element;
  }
  
  render() {
    this.container.innerHTML = '';
    
    if (this.state.isLoading) {
      const loader = this.createElement('div', ['spinner'], 'Loading...');
      this.container.appendChild(loader);
      return;
    }
    
    this.state.items.forEach(item => {
      const card = this.createElement('div', ['card']);
      const title = this.createElement('h3', ['title'], item.title);
      card.appendChild(title);
      this.container.appendChild(card);
    });
  }
}
